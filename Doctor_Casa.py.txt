def doctor_casa():
  """
  Función principal del juego Doctor Casa.
  """
  puntaje = 0
  pacientes_muertos = 0
  pacientes_salvados = 0


  # Cargar los datos del archivo
  with open("DatabaseDrCasa.txt", "r") as f:
    datos = f.read().splitlines()


  # Iterar sobre los pacientes de forma aleatoria
  while datos:
    paciente_data = random.choice(datos)
    datos.remove(paciente_data)
    paciente = paciente_data.split("D")
    relato, opciones = paciente[0], paciente[1:]


    # Mostrar el relato del paciente
    print(f"\nPaciente: {relato}")


    # Mezclar las opciones y agregar el tiempo de respuesta
    opciones_mezcladas = random.sample(opciones, len(opciones))
    tiempo_inicio = time.time()


    # Mostrar las opciones y obtener la respuesta del jugador
    for i, opcion in enumerate(opciones_mezcladas):
      print(f"{i+1}. {opcion.strip()}")
    respuesta = input("\n¿Cuál es el diagnóstico y tratamiento correctos? (1-3): ")


    # Calcular el tiempo de respuesta
    tiempo_final = time.time()
    tiempo_respuesta = tiempo_final - tiempo_inicio


    # Evaluar la respuesta del jugador
    if tiempo_respuesta > 45:
      print("\n¡Tiempo agotado! Doctor Casa, asesinaste a tu paciente.")
      pacientes_muertos += 1
    elif opciones_mezcladas[int(respuesta)-1] == opciones[0]:
      print("\n¡Doctor Casa salvaste una vida más!")
      print(f"\nCaracterísticas: {opciones[0].split('\"')[1]}")
      print(f"Tratamiento: {opciones[0].split('\"')[3]}")
      print(f"Funcionamiento del tratamiento: {opciones[0].split('\"')[5]}")
      pacientes_salvados += 1
      # Asignar puntos en base a la rapidez de la respuesta
      puntaje += max(0, 45 - int(tiempo_respuesta)) * 10
    else:
      print("\nDoctor Casa, asesinaste a tu paciente.")
      pacientes_muertos += 1


  # Mostrar resultados finales
  print(f"\n--- Fin del juego ---")
  print(f"Pacientes salvados: {pacientes_salvados}")
  print(f"Pacientes muertos: {pacientes_muertos}")
  print(f"Puntaje final: {puntaje}")


if __name__ == "__main__":
  doctor_casa()
